let cart = JSON.parse(localStorage.getItem("cart")) || []; function add(name, price){ cart.push({name, price}); saveCart(); renderCart(); } function removeItem(index){ cart.splice(index, 1); saveCart(); renderCart(); } function saveCart(){ localStorage.setItem("cart", JSON.stringify(cart)); } function renderCart(){ let items = document.getElementById("items"); let total = 0; items.innerHTML = ""; cart.forEach((item, index) => { total += item.price; items.innerHTML += `
${item.name} - KSh ${item.price} X
`; }); document.getElementById("total").innerText = total; } function checkout(){ if(cart.length === 0){ alert("Your cart is empty!"); return; } localStorage.setItem("receipt", JSON.stringify(cart)); localStorage.setItem("total", getTotal()); window.location.href = "receipt.html"; } function getTotal(){ return cart.reduce((sum, item) => sum + item.price, 0); } renderCart();
